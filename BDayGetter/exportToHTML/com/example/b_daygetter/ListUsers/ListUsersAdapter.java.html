68   	private void init_Data_B_day_countdown(TextView textView, User user) {
69   		
70   		Activity curentActivity = (Activity) super.getContext();
71   		
72   		AtomicReference<Integer> day = new AtomicReference<>(user.getDateDay());
73   		
74   		
75   		Thread thread = new Thread(() -> {
76   			try {
77   				for (int i = 0; i < 100; i++) {
78   					Thread.sleep(1000);
79   					day.getAndSet(day.get() + 1);
80   					curentActivity.runOnUiThread(() -> {
81   						textView.setText(
82   								day.toString()
83   						);
84   					});
85   				}
86   			} catch (InterruptedException ignored) {
87   			}
88   		});
89   		
90   		thread.start();
91   	}
