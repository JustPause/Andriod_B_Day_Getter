74   	public void UpdateInt() {
75   		
76   		var.nowTimeYear = java.time.LocalDate.now().getYear();
77   		var.nowTimeMonth = java.time.LocalDate.now().getMonthValue();
78   		var.nowTimeDay = java.time.LocalDate.now().getDayOfMonth();
79   		var.bDayOf = java.time.LocalDate.of(user.getDateYear(),
80   						user.getDateMonth(),
81   						user.getDateDay())
82   				.getDayOfYear();
83   		var.todayDay = java.time.LocalDate.now().getDayOfYear();
84   		var.todayTimeH = java.time.LocalDateTime.now().getHour();
85   		var.todayTimeM = java.time.LocalDateTime.now().getMinute();
86   		var.todayTimeS = java.time.LocalDateTime.now().getSecond();
87   	}

89   	public String B_day_countdown(int i) {
90   		if (var.bDayOf - var.todayDay < 0) {
91   			return var.bDayOf + 365 - var.todayDay + " Days " + (24 - var.todayTimeH) +
92   				   " Hour\n " +
93   				   (60 - var.todayTimeM) + " Minute " + (60 - var.todayTimeS) +
94   				   " Second ";
95   		} else {
96   			return var.bDayOf - var.todayDay + " Days " + (24 - var.todayTimeH) + " Hour\n " +
97   				   (60 - var.todayTimeM) + " Minute " + (60 - var.todayTimeS) + " Second ";
98   		}
99   	}
